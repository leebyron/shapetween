Tween::getEasing()
Description
Returns the Shaper currently being used as an easing function.
Example
import megamu.shapetween.*;

Tween ani;
void setup(){
  ani = new Tween(this, 2);
  ani.repeat();
  ani.setEasing( Tween.COSINE );
}

void draw(){
  background(255);
  
  ellipse(ani.time()*width, ani.position()*height, 4, 4);
}

void mousePressed(){
  Shaper ease = ani.getEasing();
  ease.clamp();
}
